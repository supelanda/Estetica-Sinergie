Setup Supabase — SQL e istruzioni (solo sviluppatore)

Esegui questo SQL nel pannello Supabase → SQL Editor per creare la tabella e configurare Row Level Security. Poi inserisci le tue credenziali nelle costanti in cima allo script di questa pagina e in sinergie-dove-siamo.html. 

-- ══════════════════════════════════════════
-- Tabella domande (richieste clienti)
-- ══════════════════════════════════════════
create table domande (
  id          uuid default gen_random_uuid() primary key,
  created_at  timestamptz default now(),
  nome        text not null,
  telefono    text,
  email       text,
  servizio    text,
  messaggio   text not null,
  stato       text default 'nuova'
              check (stato in ('nuova','letta','risposta')),
  risposta    text,
  risposta_at timestamptz
);

-- Abilita RLS
alter table domande enable row level security;

-- Il form pubblico può solo inserire (anon key)
create policy "insert_public" on domande
  for insert to anon with check (true);

-- Solo utenti autenticati possono leggere e modificare
create policy "all_authenticated" on domande
  for all to authenticated using (true);

-- Indice per data (ordinamento)
create index idx_domande_created on domande(created_at desc);
